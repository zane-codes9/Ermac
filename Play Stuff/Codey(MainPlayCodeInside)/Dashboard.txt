    <script>
        // --- CONSTANTS ---
        const N_SCALE = 20; 
        const BASE_OFFSET = 30;

        // --- DOM ELEMENTS ---
        const canvas = document.getElementById('phaseCanvas');
        const ctx = canvas.getContext('2d');
        
        const inputs = {
            phi: { num: document.getElementById('num-phi'), rng: document.getElementById('rng-phi') },
            alpha: { num: document.getElementById('num-alpha'), rng: document.getElementById('rng-alpha') },
            t: { num: document.getElementById('num-t'), rng: document.getElementById('rng-t') }
        };

        const outputs = {
            tcrit: document.getElementById('out-tcrit'),
            delta: document.getElementById('out-delta')
        };

        // --- RESIZE CANVAS ---
        // Keeps the canvas crisp on high DPI displays
        function resizeCanvas() {
            const rect = canvas.getBoundingClientRect();
            canvas.width = rect.width * 2;
            canvas.height = rect.height * 2;
            ctx.scale(2, 2);
            draw();
        }
        window.addEventListener('resize', resizeCanvas);

        // --- SYNC INPUTS ---
        function bindInput(group) {
            group.num.addEventListener('input', () => { group.rng.value = group.num.value; draw(); });
            group.rng.addEventListener('input', () => { group.num.value = group.rng.value; draw(); });
        }
        Object.values(inputs).forEach(bindInput);

        // --- PHYSICS & DRAW ENGINE ---
        function draw() {
            // 1. Get Values
            const phi = parseFloat(inputs.phi.num.value);
            const alpha = parseFloat(inputs.alpha.num.value);
            const T = parseFloat(inputs.t.num.value);

            // 2. Compute Physics
            // The Wall position
            const T_crit = (phi / (alpha * N_SCALE)) + BASE_OFFSET;
            // The Non-Linear onset
            const T_warn = T_crit - 1.5;
            
            const delta = T_crit - T;

            // 3. Update Text
            outputs.tcrit.innerText = T_crit.toFixed(2);
            outputs.delta.innerText = delta.toFixed(2);
            
            // Color Logic
            const isCrit = T >= T_crit;
            const isWarn = T >= T_warn;
            const activeColor = isCrit ? '#ff0000' : (isWarn ? '#ff0000' : '#000000'); // Red if warning or crit
            
            outputs.delta.style.color = activeColor;


            // 4. DRAW GRAPH
            const width = canvas.width / 2;
            const height = canvas.height / 2;
            
            // Clear
            ctx.clearRect(0, 0, width, height);

            // Define Scale (X Axis Range: 20 to 80)
            const minX = 20;
            const maxX = 80;
            const range = maxX - minX;
            
            const mapX = (val) => ((val - minX) / range) * width;

            // -- DRAW ZONES --
            
            // Position of The Wall
            const xWall = mapX(T_crit);
            const xWarn = mapX(T_warn);
            const xCurrent = mapX(T);

            // 1. The Critical Line (T_crit)
            ctx.beginPath();
            ctx.moveTo(xWall, 0);
            ctx.lineTo(xWall, height);
            ctx.strokeStyle = "#000";
            ctx.lineWidth = 2;
            ctx.stroke();

            // 2. The Warning Line (Dashed)
            if (xWarn > 0) {
                ctx.beginPath();
                ctx.setLineDash([5, 5]);
                ctx.moveTo(xWarn, 0);
                ctx.lineTo(xWarn, height);
                ctx.strokeStyle = "#ff0000"; // Red dashed
                ctx.lineWidth = 1;
                ctx.stroke();
                ctx.setLineDash([]); // Reset
            }

            // 3. Current Temperature (Bar)
            ctx.fillStyle = activeColor;
            // Draw a rectangle from 0 to Current T
            // Note: We clamp it visually so it doesn't fly off screen
            const drawT = Math.max(0, Math.min(xCurrent, width));
            
            // Draw a solid bar
            ctx.globalAlpha = 0.2; // light opacity fill
            ctx.fillRect(0, 0, drawT, height);
            
            // Draw the leading edge solid
            ctx.globalAlpha = 1.0;
            ctx.beginPath();
            ctx.moveTo(drawT, 0);
            ctx.lineTo(drawT, height);
            ctx.strokeStyle = activeColor;
            ctx.lineWidth = 4;
            ctx.stroke();

            // 4. Mathematical "Singularity" Curve (Cosmetic Visualization of load)
            // Just to show the exponential nature visually
            ctx.beginPath();
            ctx.strokeStyle = "#999";
            ctx.lineWidth = 1;
            
            for (let i = 0; i < width; i+=2) {
                // simple 1/x curve approaching xWall
                // normalized x (0 to 1) relative to wall
                let xVal = (i / width) * range + minX;
                if (xVal >= T_crit) break;
                
                let dist = T_crit - xVal;
                let y = height - (1 / dist) * 100; // Arbitrary scaling for visual curve
                
                if (i===0) ctx.moveTo(i, y);
                else ctx.lineTo(i, y);
            }
            ctx.stroke();
        }

        // Init
        resizeCanvas();
    </script>
